#!/bin/sh

# Check if at least one file was provided
if [ $# -eq 0 ]; then
    printf "Usage: %s <file1> <file2> ...\n" "$0"
    exit 1
fi

for f in "$@"; do
    # Verify the file exists before processing
    if [ -f "$f" ]; then
        # ffmpeg outputs 'MD5=hash', so we cut at the '='
        # hash=$(ffmpeg -loglevel error -i "$f" -vn -map 0:a -f md5 - | cut -d'=' -f2)
        hash=$(ffmpeg -loglevel error -i "$f" -vn -map 0:a -c:a pcm_s32le -f md5 - | cut -d'=' -f2)
        printf "%s  %s\n" "$hash" "$f"
    else
        printf "Error: '%s' is not a valid file\n" "$f" >&2
    fi
done
